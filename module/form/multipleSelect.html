js代码


            //这部分代码用来生成数据
            var $multipleSelect = $('#multipleSelect');
            var arr = [ { text:'MR zhang', value:'1' }, { text:'MR wang', value:'2' }, { text:'MR li', value:'3' }  ];
            var domArr = [];
            $.each(arr, function(i, item){
                var $option = $('<option></option>');
                $option.val(item.value);
                $option.html(item.text);
                domArr.push($option);
            });
            $multipleSelect.append(domArr);

            //这部分代码启动multiple select
            $("select").multipleSelect();
            $("#setSelectsBtn").click(function() {
                $("select").multipleSelect("setSelects", [1, 3]);
            });
            $("#getSelectsBtn").click(function() {
                alert("Selected values: " + $("select").multipleSelect("getSelects"));
                alert("Selected texts: " + $("select").multipleSelect("getSelects", "text"));
            });
